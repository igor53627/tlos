9const BATCH_SIZE: u32 = 128;

11/// All data needed to deploy a TLOS contract on-chain.
12#[derive(Clone, Debug)]
13pub struct TLOSDeployment {
14    pub circuit_data: Vec<u8>,
15    pub num_wires: u8,
16    pub num_gates: u32,
17    pub expected_output_hash: [u8; 32],
18    pub circuit_seed: [u8; 32],
19    /// Full-rank wire binding output: 64 x u16 = 1024 bits, stored as 4 x uint256
20    pub expected_binding_output: BindingOutput,
21}
22
23/// Generates complete TLOS deployment data from a secret and seed.
24pub fn generate_tlos(secret: [u8; 32], circuit_seed: u64) -> TLOSDeployment {
25    let config = SixSixConfig::new(circuit_seed);
26    let circuit = create_six_six_circuit(&config);
27
28    let lwe_secret = derive_secret(secret);
29    let mut rng = ChaCha20Rng::seed_from_u64(circuit_seed.wrapping_add(0x12345678));
30
31    let mut circuit_data = Vec::new();
32    for gate in &circuit.gates {
33        let encoded = encode_gate(gate, &lwe_secret, &mut rng);
34        circuit_data.extend_from_slice(&encoded);
35    }
36
37    // Convert circuit_seed u64 to bytes32 (big-endian, right-aligned)
38    let mut circuit_seed_bytes = [0u8; 32];
39    circuit_seed_bytes[24..32].copy_from_slice(&circuit_seed.to_be_bytes());
40
41    let mask = if circuit.num_wires >= 64 {
42        u64::MAX
43    } else {
44        (1u64 << circuit.num_wires) - 1
45    };
46    // Solidity: wires = uint256(input) & mask
47    // input is bytes32, low bits are in the LAST 8 bytes (big-endian uint256)
48    let initial_wires = u64::from_be_bytes(secret[24..32].try_into().unwrap()) & mask;
49
50    let (final_wires, binding_output) =
51        simulate_evaluation(&circuit, initial_wires, circuit_seed_bytes);
52
53    // Solidity: keccak256(abi.encodePacked(wires))
54    // wires is uint256, so we encode as 32-byte big-endian
55    let expected_output_hash = {
56        let mut hasher = Keccak256::new();
57        let mut wires_bytes = [0u8; 32];
58        wires_bytes[24..32].copy_from_slice(&final_wires.to_be_bytes());
59        hasher.update(&wires_bytes);
60        let result = hasher.finalize();
61        let mut out = [0u8; 32];
62        out.copy_from_slice(&result);
63        out
64    };
65
66    TLOSDeployment {
67        circuit_data,
68        num_wires: circuit.num_wires as u8,
69        num_gates: circuit.gates.len() as u32,
70        expected_output_hash,
71        circuit_seed: circuit_seed_bytes,
72        expected_binding_output: binding_output,
73    }
74}

76fn simulate_evaluation(
77    circuit: &crate::circuit::Circuit,
78    initial_wires: u64,
79    circuit_seed: [u8; 32],
80) -> (u64, BindingOutput) {
81    let mut wires = initial_wires;
82    let num_gates = circuit.gates.len() as u32;
83    let num_wires = circuit.num_wires as u8;
84
85    // Mirrors TLOS.sol: bindingAcc = _wireBindingHash(wires, 0)
86    let mut binding_acc = wire_binding_init(wires, num_wires, circuit_seed);
87
88    let mut batch_start = 0u32;
89    while batch_start < num_gates {
90        let batch_end = std::cmp::min(batch_start + BATCH_SIZE, num_gates);
91
92        for gate_idx in batch_start..batch_end {
93            let gate = &circuit.gates[gate_idx as usize];
94            let c1_val = (wires >> gate.control1()) & 1 != 0;
95            let c2_val = (wires >> gate.control2()) & 1 != 0;
96            let tt_bit = gate.truth_table_bit(c1_val, c2_val);
97            let active_bit = (wires >> gate.active()) & 1 != 0;
98            let new_bit = active_bit ^ tt_bit;
99            if new_bit {
100                wires |= 1 << gate.active();
101            } else {
102                wires &= !(1 << gate.active());
103            }
104        }
105
106        // Mirrors TLOS.sol: combined = bindingAcc[0..3] ^ wires; bindingAcc = _wireBindingHash(combined, batchEnd)
107        binding_acc = wire_binding_update(binding_acc, batch_end, wires, num_wires, circuit_seed);
108        batch_start = batch_end;
109    }
110
111    (wires, binding_acc)
112}

114#[cfg(test)]
115mod tests {
116    use super::*;
117
118    #[test]
119    fn test_generate_tlos() {
120        use crate::lwe::GATE_SIZE;
121        let secret = [0x42u8; 32];
122        let deployment = generate_tlos(secret, 12345);
123        assert_eq!(deployment.num_wires, 64);
124        assert_eq!(deployment.num_gates, 640);
125        assert_eq!(deployment.circuit_data.len(), 640 * GATE_SIZE);
126        // Wire binding output should have content in at least one word
127        assert!(deployment.expected_binding_output.is_nonzero());
128    }
129
130    #[test]
131    fn test_deterministic_generation() {
132        let secret = [0x55u8; 32];
133        let d1 = generate_tlos(secret, 999);
134        let d2 = generate_tlos(secret, 999);
135        assert_eq!(d1.circuit_data, d2.circuit_data);
136        assert_eq!(d1.expected_output_hash, d2.expected_output_hash);
137        assert_eq!(d1.expected_binding_output, d2.expected_binding_output);
138        assert_eq!(d1.circuit_seed, d2.circuit_seed);
139    }
140
141    #[test]
142    fn test_binding_output_is_1024_bit() {
143        let secret = [0xABu8; 32];
144        let deployment = generate_tlos(secret, 42);
145        // 1024-bit wire binding output: 4 x 256-bit words
146        // At least one word should be non-zero
147        assert!(deployment.expected_binding_output.is_nonzero());
148    }
