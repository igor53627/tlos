3/// LWE modulus (largest 16-bit prime), shared with Layer 2.
4pub const Q: u32 = 65521;
5
6/// Number of rows in the wire binding matrix (64 wires).
7pub const BINDING_ROWS: usize = 64;
8
9/// Compute full-rank wire binding hash H(x) = A * x mod q where A is 64x64
10/// Mirrors TLOS.sol::_wireBindingHash exactly
11///
12/// - `input_bits`: lower `num_wires` bits contain the wire / combined state  
13/// - `gate_idx`: gate index (0 for init, batchEnd for updates)
14/// - `num_wires`: 1..=64
15/// - `circuit_seed`: 32-byte seed matching Solidity `circuitSeed`
16/// - Returns: `[U256; 4]` packed as 64 x u16 (1024 bits total, 16 x 16-bit per U256)
17///
18/// Optimized: derives 16 coefficients per keccak (320 calls vs 4096).
19/// Note: We use `[u128; 2]` to represent each uint256 (low, high).
20pub fn wire_binding_hash(
21    input_bits: u64,
22    gate_idx: u32,
23    num_wires: u8,
24    circuit_seed: [u8; 32],
25) -> BindingOutput {
26    let mut output = BindingOutput::default();
27
28    for row in 0..BINDING_ROWS {
29        // rowSeed = keccak256(seed, gateIdx, row)
30        // Solidity packs: seed (32 bytes) + gateIdx (32 bytes) + row (32 bytes)
31        let mut hasher = Keccak256::new();
32        hasher.update(&circuit_seed);
33        
34        let mut gate_bytes = [0u8; 32];
35        gate_bytes[28..32].copy_from_slice(&gate_idx.to_be_bytes());
36        hasher.update(&gate_bytes);
37        
38        let mut row_bytes = [0u8; 32];
39        row_bytes[28..32].copy_from_slice(&(row as u32).to_be_bytes());
40        hasher.update(&row_bytes);
41        
42        let row_seed: [u8; 32] = hasher.finalize().into();
43
44        let mut sum: u32 = 0;
45        let mut col: u32 = 0;
46        let mut block_idx: u32 = 0;
47        
48        // Process coefficients in blocks of 16 (each keccak gives 16 x u16)
49        while col < (num_wires as u32) {
50            // blockDigest = keccak256(rowSeed, blockIdx)
51            let mut block_hasher = Keccak256::new();
52            block_hasher.update(&row_seed);
53            
54            let mut block_bytes = [0u8; 32];
55            block_bytes[28..32].copy_from_slice(&block_idx.to_be_bytes());
56            block_hasher.update(&block_bytes);
57            
58            let block_digest: [u8; 32] = block_hasher.finalize().into();
59            
60            // Extract up to 16 coefficients from this block
61            // Solidity treats keccak output as uint256 (big-endian bytes -> single 256-bit int)
62            // shr(k*16, blockDigest) extracts the k-th u16 starting from low bits
63            // bytes[0..16] = high 128 bits, bytes[16..32] = low 128 bits
64            let low_u128 = u128::from_be_bytes(block_digest[16..32].try_into().unwrap());
65            let high_u128 = u128::from_be_bytes(block_digest[0..16].try_into().unwrap());
66            
67            for k in 0..16u32 {
68                if col >= (num_wires as u32) {
69                    break;
70                }
71                
72                // Extract u16 at position k (from the 256-bit value)
73                // k=0..7 are in low_u128, k=8..15 are in high_u128
74                let shift = (k % 8) * 16;
75                let aij = if k < 8 {
76                    ((low_u128 >> shift) & 0xFFFF) % (Q as u128)
77                } else {
78                    ((high_u128 >> shift) & 0xFFFF) % (Q as u128)
79                };
80                let aij = aij as u32;
81                
82                let bit_val = ((input_bits >> col) & 1) as u32;
83                
84                if bit_val == 1 {
85                    sum += aij;
86                    if sum >= Q {
87                        sum -= Q;
88                    }
89                }
90                
91                col += 1;
92            }
93            
94            block_idx += 1;
95        }
96
97        // Pack into output: 16 elements per uint256 (16 * 16 bits = 256 bits)
98        // Each uint256 is stored as [low_u128, high_u128]
99        let word_idx = row / 16;
100        let elem_in_word = row % 16;
101        let bit_pos = elem_in_word * 16;
102        
103        if bit_pos < 128 {
104            output.words[word_idx].0 |= (sum as u128) << bit_pos;
105        } else {
106            output.words[word_idx].1 |= (sum as u128) << (bit_pos - 128);
107        }
108    }
109
110    output
111}

113/// Represents 4 x uint256 (each as low/high u128 pair)
114#[derive(Clone, Copy, Debug, Default, PartialEq, Eq)]
115pub struct BindingOutput {
116    pub words: [(u128, u128); 4],
117}
118
119impl BindingOutput {
120    pub fn is_nonzero(&self) -> bool {
121        self.words.iter().any(|(lo, hi)| *lo != 0 || *hi != 0)
122    }
123    
124    pub fn xor_all(&self) -> u128 {
125        let mut result = 0u128;
126        for (lo, hi) in &self.words {
127            result ^= lo ^ hi;
128        }
129        result
130    }
131}

135/// Initialize wire binding accumulator with initial wire state
136/// Mirrors: bindingAcc = _wireBindingHash(wires, 0)
137pub fn wire_binding_init(
138    wires: u64,
139    num_wires: u8,
140    circuit_seed: [u8; 32],
141) -> BindingOutput {
142    wire_binding_hash(wires, 0, num_wires, circuit_seed)
143}
144
145/// Update wire binding accumulator after a batch of gates
146/// Mirrors Solidity: `combined = bindingAcc[0] ^ bindingAcc[1] ^ bindingAcc[2] ^ bindingAcc[3] ^ wires`
147///                   bindingAcc = _wireBindingHash(combined, batchEnd)
148pub fn wire_binding_update(
149    acc: BindingOutput,
150    batch_end: u32,
151    wires: u64,
152    num_wires: u8,
153    circuit_seed: [u8; 32],
154) -> BindingOutput {
155    // XOR all uint256 words together, then XOR with wires
156    let combined = acc.xor_all() ^ (wires as u128);
157    // Use only the low 64 bits for the next hash input
158    wire_binding_hash(combined as u64, batch_end, num_wires, circuit_seed)
159}

165    #[test]
166    fn test_wire_binding_init_deterministic() {
167        let seed = [0x42u8; 32];
168        let wires = 0x123456789ABCDEFu64;
169        let h1 = wire_binding_init(wires, 64, seed);
170        let h2 = wire_binding_init(wires, 64, seed);
171        assert_eq!(h1, h2);
172    }
173
174    #[test]
175    fn test_wire_binding_update_deterministic() {
176        let seed = [0x55u8; 32];
177        let acc = wire_binding_init(0x1234567890ABCDEFu64, 64, seed);
178        let wires = 0xDEADBEEFCAFEBABEu64;
179        let h1 = wire_binding_update(acc, 64, wires, 64, seed);
180        let h2 = wire_binding_update(acc, 64, wires, 64, seed);
181        assert_eq!(h1, h2);
182    }
183
184    #[test]
185    fn test_wire_binding_chain() {
186        let seed = [0x01u8; 32];
187        let initial_wires = 0xFFFFu64;
188        let acc = wire_binding_init(initial_wires, 64, seed);
189        let acc = wire_binding_update(acc, 64, 0x1234, 64, seed);
190        let acc = wire_binding_update(acc, 128, 0x5678, 64, seed);
191        assert!(acc.is_nonzero());
192    }
