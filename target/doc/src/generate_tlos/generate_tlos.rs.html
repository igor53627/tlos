1use clap::Parser;
2use tlos::generate_tlos;
3
4#[derive(Parser, Debug)]
5#[command(name = "generate_tlos")]
6#[command(about = "Generate TLOS deployment data")]
7struct Args {
8    #[arg(long, help = "Secret as 64-character hex string")]
9    secret: String,
10
11    #[arg(long, default_value = "0", help = "Circuit seed")]
12    seed: u64,
13
14    #[arg(long, help = "Output circuit data to file")]
15    output: Option<String>,
16}
17
18fn main() {
19    let args = Args::parse();
20
21    let secret_hex = args.secret.strip_prefix("0x").unwrap_or(&args.secret);
22    let secret_bytes = hex::decode(secret_hex).expect("Invalid hex for secret");
23    if secret_bytes.len() != 32 {
24        eprintln!("Error: secret must be exactly 32 bytes (64 hex chars)");
25        std::process::exit(1);
26    }
27    let secret: [u8; 32] = secret_bytes.try_into().unwrap();
28
29    let deployment = generate_tlos(secret, args.seed);
30
31    println!("TLOS Deployment Generated");
32    println!("========================");
33    println!("Wires: {}", deployment.num_wires);
34    println!("Gates: {}", deployment.num_gates);
35    println!("Circuit data size: {} bytes", deployment.circuit_data.len());
36    println!(
37        "Expected output hash: 0x{}",
38        hex::encode(deployment.expected_output_hash)
39    );
40    println!(
41        "Circuit seed: 0x{}",
42        hex::encode(deployment.circuit_seed)
43    );
44    println!("Expected wire binding output (4 x u256, as lo|hi u128 pairs):");
45    for (i, (lo, hi)) in deployment.expected_binding_output.words.iter().enumerate() {
46        println!("  [{i}]: lo=0x{lo:032x} hi=0x{hi:032x}");
47    }
48
49    if let Some(output_path) = args.output {
50        std::fs::write(&output_path, &deployment.circuit_data)
51            .expect("Failed to write circuit data");
52        println!("Circuit data written to: {}", output_path);
53    } else {
54        println!("\nCircuit data (hex, first 200 bytes):");
55        println!("{}", hex::encode(&deployment.circuit_data[..200]));
56    }
57}
